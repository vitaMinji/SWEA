#include<iostream>

using namespace std;

int main(void){
    int tc;
    cin>>tc;
    long long sheep;
    
    for(int i=0; i<tc; i++){
        cin>>sheep;
        int memory[10]={0};
        int answer = 1; //ë‹µ
        long long never = sheep;
        int temp;
        
        for(;;){
            temp = sheep*answer;
            bool check = true;
            while(1){
                memory[temp%10] = 1;
                temp = temp/10;
                if(temp<1) break;
            }
          
            for(int j = 0; j<10; j++){
                if(memory[j]!=1){
                    check = false;
                    break;
                }
            }
            
            if(check==true) break;
            else answer++;
        }
        cout<<"#"<<i+1<<" "<<answer*never<<endl;
    }
}
